``` r
compareLifeExp <- gapminder %>% 
  select(year, country, lifeExp) %>%
  filter(country == 'Norway' | country == 'Canada' | country == 'Sudan') %>%
  group_by(year) %>%
  spread(key = country, value=lifeExp)
```
